Simple Obstacle Avoidance Robot

A simple obstacle avoidance robot using Arduino and ultrasonic sensors.

ArduinoUltrasonic Sensors 

// Simple Arduino Obstacle Avoidance Code
const int trigPin = 9;
const int echoPin = 10;
const int motorA1 = 5;
int echo=7;
int trig=8;
float dur,dis;
int ML1=13;
int ML2=11;
int MR1=10;
int MR2=2;
int ENA=5;
int ENB=6;
int speed=150;
int buzzer=3;

void setup(){
  pinMode(trig,OUTPUT);
  pinMode(echo,INPUT);
  Serial.begin(9600);
}
void ultrasonic(){
  digitalWrite(trig,LOW);
  delay(10);
  digitalWrite(trig,HIGH);
  delay(10);
  digitalWrite(trig,LOW);
  dur=pulseIn(echo,HIGH);
  dis=(dur*0.034)/2;
}
void forward(){
  analogWrite(ENA,speed);
  analogWrite(ENB,speed);
  digitalWrite(ML1,HIGH);
  digitalWrite(ML2,LOW);
  digitalWrite(MR1,HIGH);
  digitalWrite(MR2,LOW);
}
void stop(){
  analogWrite(ENA,LOW);
  analogWrite(ENB,LOW);
  digitalWrite(ML1,LOW);
  digitalWrite(ML2,LOW);
  digitalWrite(MR1,LOW);
  digitalWrite(MR2,LOW);
}
void backward(){
  analogWrite(ENA,speed);
  analogWrite(ENB,speed);
  digitalWrite(ML1,LOW);
  digitalWrite(ML2,HIGH);
  digitalWrite(MR1,LOW);
  digitalWrite(MR2,HIGH);
}
void loop(){
  ultrasonic();
  if(dis<15){
    stop();
    backward();
    tone(buzzer,500);
  }else{
    forward();
    noTone(buzzer);
  }
  Serial.println(dis);
}
